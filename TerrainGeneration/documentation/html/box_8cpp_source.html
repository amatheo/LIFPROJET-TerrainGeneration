134void Box::SetParallelepipedic(double size, int& x, int& y, int& z)
135{
136 // Diagonal
137 Vector d = (b - a);
138
139 // Integer sizes
140 // Bug tracking: adding 0.99 avoids keeping track of which indexes are the maxima
141 x = int(d[0] / size + 0.99);
142 y = int(d[1] / size + 0.99);
143 z = int(d[2] / size + 0.99);
144
145 // Expand if necessary
146 if (x == 0) { x++; }
147 if (y == 0) { y++; }
148 if (z == 0) { z++; }
149
150 // Center
151 Vector c = 0.5 * (a + b);
152
153 // Diagonal
154 Vector e = Vector(x, y, z) * size / 2.0;
155 a = c - e;
156 b = c + e;
157}
