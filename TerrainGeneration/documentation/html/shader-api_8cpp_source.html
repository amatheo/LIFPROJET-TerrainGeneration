28static
29std::string prepare_source(std::string file, const std::string& definitions)
30{
31 if (file.empty())
32 return std::string();
33
34 // un peu de gymnastique, #version doit rester sur la premiere ligne, meme si on insere des #define dans le source
35 std::string source;
36
37 // recupere la ligne #version
38 std::string version;
39 size_t b = file.find("#version");
40 if (b != std::string::npos)
41 {
42 size_t e = file.find('\n', b);
43 if (e != std::string::npos)
44 {
45 version = file.substr(0, e + 1);
46 file.erase(0, e + 1);
47
48 if (file.find("#version") != std::string::npos)
49 {
50 printf("[error] found several #version directives. failed.\n");
51 return std::string();
52 }
53 }
54 }
55 else
56 {
57 printf("[error] no #version directive found. failed.\n");
58 return std::string();
59 }
60
61 // reconstruit le source complet
62 if (definitions.empty() == false)
63 {
64 source.append(version); // insere la version
65 source.append(definitions).append("\n"); // insere les definitions
66 source.append(file); // insere le source
67 }
68 else
69 {
70 source.append(version); // re-insere la version (supprimee de file)
71 source.assign(file); // insere le source
72 }
73
74 return source;
75}

130GLuint compile_shader(const GLuint program, const GLenum shader_type, const std::string& source)
131{
132 if (source.size() == 0 || shader_type == 0)
133 return 0;
134
135 GLuint shader = glCreateShader(shader_type);
136 glAttachShader(program, shader);
137
138 const char* sources = source.c_str();
139 glShaderSource(shader, 1, &sources, NULL);
140 glCompileShader(shader);
141
142 GLint status;
143 glGetShaderiv(shader, GL_COMPILE_STATUS, &status);
144 return (status == GL_TRUE) ? shader : 0;
145}

148int reload_program(GLuint program, const char* filename, const char* definitions)
149{
150 if (program == 0)

152

154 int shaders_max = 0;

156 if (shaders_max > 0)
157 {
158 std::vector<GLuint> shaders(shaders_max, 0);
159 glGetAttachedShaders(program, shaders_max, NULL, &shaders.front());
160 for (int i = 0; i < shaders_max; i++)
161 {
162 glDetachShader(program, shaders[i]);
163 glDeleteShader(shaders[i]);
164 }
165 }
166
167#ifdef GL_VERSION_4_3
168 glObjectLabel(GL_PROGRAM, program, -1, filename);
169#endif
170
171 // prepare les sources
172 std::string common_source = read(filename);
173 for (int i = 0; i < shader_keys_max; i++)
174 {
175 if (common_source.find(shader_keys[i]) != std::string::npos)
176 {
177 // cree et compile les shaders detectes dans le source
178 std::string source = prepare_source(common_source, std::string(definitions).append("#define ").append(shader_keys[i]).append("\n"));
179 GLuint shader = compile_shader(program, shader_types[i], source);
180 if (shader == 0)
181 printf("[error] compiling %s...\n%s\n", shader_string(shader_types[i]), definitions);
182 }
183 }
184
185 // linke les shaders
186 glLinkProgram(program);
187
188 // verifie les erreurs
189 GLint status;
190 glGetProgramiv(program, GL_LINK_STATUS, &status);
191 if (status == GL_FALSE)
192 {
193 printf("[error] linking program %u '%s'...\n", program, filename);
194 return -1;
195 }
196
197 // pour etre coherent avec les autres fonctions de creation, active l'objet gl qui vient d'etre cree.
198 glUseProgram(program);
199 return 0;

238void print_line(std::string& errors, const char* source, const int begin_id, const int line_id)
239{
240 int line = 0;
241 char last = '\n';
242 for (unsigned int i = 0; source[i] != 0; i++)
243 {
244 if (line > line_id)
245 break;
246
247 if (last == '\n')
248 {
249 line++;
250 if (line >= begin_id && line <= line_id)
251 {
252 errors.append(" ");
253 errors.push_back('0' + (line / 1000) % 10);

255 errors.push_back('0' + (line / 10) % 10);
256 errors.push_back('0' + (line / 1) % 10);
257 errors.append(" ");
258 }
259 }
260
261 if (line >= begin_id && line <= line_id)
262 {
263 if (source[i] == '\t')
264 errors.append(" ");
265 else
266 errors.push_back(source[i]);
267 }

296 print_line(errors, source, last_line + 1, line_id);
297 errors.append("\n");
298 }
299 }

301 for (i += position; log[i] != 0; i++)

303 errors.push_back(log[i]);

306 }
