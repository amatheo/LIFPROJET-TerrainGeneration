62bool Triangle::Intersect(const Ray& ray, double& t, double& u, double& v) const
63{
64 Vector e[2];
65
66 // Find edge vectors
67 e[0] = p[1] - p[0];
68 e[1] = p[2] - p[0];
69
70 Vector pvec = ray.Direction() / e[1];
71
72 // Determinant, if determinant is near zero, ray lies in plane of triangle
73 double det = e[0] * pvec;
74
75 if ((det > -epsilon) && (det < epsilon))
76 return false;
77 det = 1.0 / det;
78
79 // calculate distance from first vertex to ray origin
80 Vector tvec = ray.Origin() - p[0];
81
82 // calculate U parameter and test bounds
83 u = (tvec * pvec) * det;
84 if ((u < 0.0) || (u > 1.0))
85 return false;
86
87 // calculate V parameter and test bounds
88 Vector qvec = tvec / e[0];
89
90 v = (ray.Direction() * qvec) * det;
91 if ((v < 0.0) || (u + v) > 1.0)
92 return false;
93
94 // Ray intersects triangle
95 t = (e[1] * qvec) * det;
96 return true;
97}
