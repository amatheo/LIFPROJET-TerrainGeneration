162/*
163\brief Check if a point is inside or outside of the object.
164
165This is a brute force algorithm, that computes all the intersections between the ray and the triangles of the object.
166
167The mesh should be closed, i.e. without holes.
168\param p Point.
169*/
170bool Mesh::Inside(const Vector& p) const
171{
172 // Ray used for intersecting the triangles
173 const Ray ray(p, Vector(1.0, 0.0, 0.0));
174 bool check = false;
175
176 for (int i = 0; i < varray.size() / 3; i++)
177 {
178 // Temporary depth
179 double t;
180 double u, v;
181 // Edge
182 if (GetTriangle(i).Intersect(ray, t, u, v))
183 {
184 if (t >= 0.0) check = !check;
185 }
186 }
187
188 return check;
189}

246void Mesh::Load(const QString& filename)
247{
248 vertices.clear();
249 normals.clear();
250 varray.clear();
251 narray.clear();
252
253 QFile data(filename);
254
255 if (!data.open(QFile::ReadOnly))
256 return;
257 QTextStream in(&data);
258
259 // Set of regular expressions : Vertex, Normal, Triangle
260 QRegExp rexv("v\\s*([-|+|\\s]\\d*\\.\\d+)\\s*([-|+|\\s]\\d*\\.\\d+)\\s*([-|+|\\s]\\d*\\.\\d+)");
261 QRegExp rexn("vn\\s*([-|+|\\s]\\d*\\.\\d+)\\s*([-|+|\\s]\\d*\\.\\d+)\\s*([-|+|\\s]\\d*\\.\\d+)");
262 QRegExp rext("f\\s*(\\d*)/\\d*/(\\d*)\\s*(\\d*)/\\d*/(\\d*)\\s*(\\d*)/\\d*/(\\d*)");
263 while (!in.atEnd())
264 {
265 QString line = in.readLine();
266 if (rexv.indexIn(line, 0) > -1)
267 {
268 Vector q = Vector(rexv.cap(1).toDouble(), rexv.cap(2).toDouble(), rexv.cap(3).toDouble()); vertices.append(q);
269 }
270 else if (rexn.indexIn(line, 0) > -1)
271 {
272 Vector q = Vector(rexn.cap(1).toDouble(), rexn.cap(2).toDouble(), rexn.cap(3).toDouble()); normals.append(q);
273 }
274 else if (rext.indexIn(line, 0) > -1)
275 {
276 varray.append(rext.cap(1).toInt() - 1);
277 varray.append(rext.cap(3).toInt() - 1);
278 varray.append(rext.cap(5).toInt() - 1);
279 narray.append(rext.cap(2).toInt() - 1);
280 narray.append(rext.cap(4).toInt() - 1);
281 narray.append(rext.cap(6).toInt() - 1);
282 }
283 }
284 data.close();
285}
