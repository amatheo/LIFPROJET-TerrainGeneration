333void MeshWidget::paintGL()
334{
335 // Custom update from user
336 emit _signalUpdate();
337
338 // Clear
339 glClearColor(1.0f, 1.0f, 1.0f, 1.f);
340 glMatrixMode(GL_MODELVIEW);
341 glLoadIdentity();
342 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
343
344 // Move camera
345 if (MoveAt)
346 {
347 if (stepAt < 15)
348 {
349 stepAt++;
350 double alpha = double(stepAt) / 15.;
351 camera.SetAt(currentAt * (1. - alpha) + toAt * (alpha));
352 }
353 else
354 MoveAt = false;
355 }
356 gluLookAt(camera.Eye()[0], camera.Eye()[1], camera.Eye()[2], camera.At()[0], camera.At()[1], camera.At()[2], camera.Up()[0], camera.Up()[1], camera.Up()[2]);
357
358 // Sky
359 glEnable(GL_DEPTH_TEST);
360 glDepthFunc(GL_LEQUAL);
361 glUseProgram(skyboxShader);
362 glBindVertexArray(skyboxVAO);
363 glUniform3f(0, camera.Eye()[0], camera.Eye()[1], camera.Eye()[2]);
364 glUniform3f(1, camera.At()[0], camera.At()[1], camera.At()[2]);
365 glUniform3f(2, camera.Up()[0], camera.Up()[1], camera.Up()[2]);
366 glUniform2f(3, width(), height());
367 glDrawArrays(GL_TRIANGLE_STRIP, 0, 4);
368
369 // Draw meshes
370 profiler.BeginGPU();
371 GLfloat ModelViewMatrix[16];
372 GLfloat ProjectionMatrix[16];
373 glGetFloatv(GL_MODELVIEW_MATRIX, ModelViewMatrix);
374 glGetFloatv(GL_PROJECTION_MATRIX, ProjectionMatrix);
375 glUseProgram(mainShaderProgram);
376
377 // Shared uniforms
378 glUniformMatrix4fv(glGetUniformLocation(mainShaderProgram, "ModelViewMatrix"), 1, 0, ModelViewMatrix);
379 glUniformMatrix4fv(glGetUniformLocation(mainShaderProgram, "ProjectionMatrix"), 1, 0, ProjectionMatrix);
380
381 for (MeshIterator i = objects.begin(); i != objects.end(); i++)
382 {
383 if (!i.value()->enabled)
384 continue;
385
386 // Uniforms
387 glUniform2f(glGetUniformLocation(mainShaderProgram, "WIN_SCALE"), width() / 2.0f, height() / 2.0f);

389 glUniform1i(glGetUniformLocation(mainShaderProgram, "useWireframe"), i.value()->useWireframe ? 1 : 0);
390 glUniform1i(glGetUniformLocation(mainShaderProgram, "material"), (int)i.value()->material);
391 glUniform1i(glGetUniformLocation(mainShaderProgram, "shading"), (int)i.value()->shading);

395 glDrawElements(GL_TRIANGLES, (GLsizei)i.value()->triangleCount, GL_UNSIGNED_INT, 0);
396 }
397 profiler.EndGPU();

746void MeshWidget::mouseMoveEvent(QMouseEvent* e)
747{
748 int x = e->globalX();
749 int y = e->globalY();
750 if ((e->modifiers() & Qt::AltModifier))
751 {
752 // Displacement mode
753 double MoveScale = Norm(camera.View()) * 0.015 * 0.05;
754 if (e->buttons() & Qt::LeftButton)
755 {
756 // Alt + Left Mouse Move : Rotation
757 camera.LeftRightRound((x0 - x) * 0.01);
758 camera.UpDownRound((y0 - y) * 0.005);
759 }
760 else if (e->buttons() & Qt::RightButton)
761 {
762 // Alt + Right Mouse Move : Forward and Backward
763 camera.BackForth((y - y0) * MoveScale);
764 QApplication::setOverrideCursor(QCursor(Qt::SplitVCursor));
765 }
766 else if (e->buttons() & Qt::MiddleButton)
767 {
768 // Alt + Left Mouse Move : Plan displacement
769 camera.LeftRightHorizontal((x - x0) * MoveScale);
770 camera.UpDownVertical((y - y0) * MoveScale);
771 QApplication::setOverrideCursor(QCursor(Qt::SizeAllCursor));
772 }
773
774 x0 = e->globalX();
775 y0 = e->globalY();
776
777 emit _signalMouseMove(e);
778 }
779 if (e->modifiers() & Qt::ControlModifier)
780 {
781 emit _signalMouseMoveEdit(e);
782
783 x0 = e->globalX();
784 y0 = e->globalY();

793{

795 {

799 cameraOrthoSize -= MoveScale;
